  async function stitchToVideo(imagePaths, audioPath) {
    return new Promise((resolve, reject) => {
      try {
        const outputPath = path.join(videoDir, `video_${Date.now()}.mp4`);
        
        let cmd = ffmpeg();

        // Add each image (5 seconds each)
        imagePaths.forEach(img => {
          cmd = cmd.input(img).inputOptions('-t 5');
        });

        // Add audio
        cmd = cmd.input(audioPath);

        // Create concat filter
        const numImages = imagePaths.length;
        let filterComplex = '';
        for (let i = 0; i < numImages; i++) {
          filterComplex += `[${i}:v]`;
        }
        filterComplex += `concat=n=${numImages}:v=1:a=0[v];[v][${numImages}:a]concat=n=1:v=1:a=1[outv][outa]`;

        cmd.outputOptions([
          '-filter_complex', filterComplex,
          '-map', '[outv]',
          '-map', '[outa]',
          '-c:v', 'libx264',
          '-preset', 'fast',
          '-c:a', 'aac'
        ])
        .output(outputPath)
        .on('end', () => {
          console.log('  ✅ Video created');
          resolve(outputPath);
        })
        .on('error', (err) => {
          console.error('FFmpeg error:', err);
          reject(err);
        })
        .run();

      } catch (error) {
        reject(error);
      }
    });
  }
